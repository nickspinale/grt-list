\thm{
    For all $k \ge 2$, there exists some graph $G$ such that $\chi(G) > k$ while $\omega(G) = 2$
}

\dfn{
    The girth of $G$, denoted $g(G)$, is the length of the shortest cycle in $G$.
}

\thm[Erd\H{o}s]{
    For all $k \in \N$, there exists some graph $G$ such that $g(G) > k$ and $\chi(G) > k$.
}

\obs{
    There is no local reason for $\chi(G)$ to be large.
}
